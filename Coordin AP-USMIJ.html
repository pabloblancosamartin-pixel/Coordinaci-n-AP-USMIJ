<script>

// Max scores per disorder (approximate full picture)

const MAX = { tdah: 14, tea: 9, toc: 6, tnd: 8, fobias: 8, dep: 11 };

// Threshold for "compatible" label

const THRESH = { tdah: 6, tea: 5, toc: 3, tnd: 4, fobias: 4, dep: 5 };

// Threshold for "probable" label

const HIGH   = { tdah: 10, tea: 7, toc: 5, tnd: 6, fobias: 6, dep: 8 };


const scores = { tdah: 0, tea: 0, toc: 0, tnd: 0, fobias: 0, dep: 0 };


function updateUI() {

  const keys = Object.keys(scores);

  const maxScore = Math.max(...keys.map(k => scores[k]), 1);


  keys.forEach(k => {

    const s = scores[k];

    const pct = Math.min(100, (s / MAX[k]) * 100);

    document.getElementById(`bar-${k}`).style.width = pct + '%';

    document.getElementById(`sc-${k}`).textContent = s.toFixed(1) + ' pts';


    const card = document.querySelector(`.dx-card[data-dx="${k}"]`);

    const lbl  = document.getElementById(`lbl-${k}`);


    card.classList.remove('active','threshold','full');


    if (s === 0) {

      lbl.textContent = 'Sin datos suficientes';

    } else if (s < THRESH[k]) {

      lbl.textContent = 'Datos insuficientes para aproximar';

      card.classList.add('active');

    } else if (s < HIGH[k]) {

      lbl.textContent = '⚑ Compatible — evaluar criterios completos';

      card.classList.add('active','threshold');

    } else {

      lbl.textContent = '▲ Alta concordancia clínica';

      card.classList.add('active','full');

    }

  });


  // Sort cards by score descending

  const sidebar = document.querySelector('.sidebar');

  const cards = [...document.querySelectorAll('.dx-card')];

  cards.sort((a,b) => scores[b.dataset.dx] - scores[a.dataset.dx]);

  const title = sidebar.querySelector('.sidebar-title');

  const btn   = sidebar.querySelector('.reset-btn');

  cards.forEach(c => sidebar.insertBefore(c, btn));

}


// Item click handling

document.querySelectorAll('.item').forEach(item => {

  item.addEventListener('click', () => {

    const wasChecked = item.classList.contains('checked');

    item.classList.toggle('checked');

    const pts = JSON.parse(item.dataset.points);

    Object.entries(pts).forEach(([k, v]) => {

      scores[k] = Math.max(0, scores[k] + (wasChecked ? -v : v));

    });

    updateUI();

  });

});


function resetAll() {

  document.querySelectorAll('.item.checked').forEach(i => i.classList.remove('checked'));

  Object.keys(scores).forEach(k => scores[k] = 0);

  updateUI();

}


updateUI();

</script>
